# NOTE, to mount the current directory, use the following:
# Windows cmd: %cd%:/app
# PowerShell: ${PWD}:/usr/app
# Linux: -v $(pwd):/app

# use a docker host bind to the disk of your e.g. laptop
docker run -itd -v $(pwd):/app node:lts-slim

docker ps
docker exec -it <container_id> /bin/bash

    # now inside the container
	cd /app
	npm install -g @vue/cli
	vue create .
        # pick your choices now or stick with the default config
        # After the installation you should see something like
            # Vue CLI v3.0.1
            # yarn install v1.9.2
    exit

docker rm -f <container_id>
